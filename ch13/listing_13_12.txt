import { City, Person, Product, Employee } from "./dataTypes" ;

function getValue<T, K extends keyof T>(item: T, keyname: K) :T[K] {
    return item[keyname];
}

type priceType = Product["price"] ;
type allTypes = Product[keyof Product];
